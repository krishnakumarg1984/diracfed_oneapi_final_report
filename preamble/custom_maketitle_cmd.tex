% https://tex.stackexchange.com/a/94679
\let \oldmaketitle = \maketitle
\def \newmaketitle {\oldmaketitle \vspace*{-50pt}\vspace{\baselineskip}}

\def \titleDecide {%
  \ifx \nextToken \section
    \let \next = \newmaketitle
  \else
    \let \next = \oldmaketitle
  \fi
  \next
}

\renewcommand{\maketitle}{\futurelet \nextToken \titleDecide}
